\begin{frame}
	\frametitle{Représentation graphique d'une règle}
	\begin{center}
	$\forall x$ (homme($x$) $\rightarrow \exists z$ (homme($z$) $\wedge$
	pere($z$,$x$)))
	\end{center}
	\begin{figure}
		\begin{tikzpicture}[
			auto,
			inner sep=1mm,
			scale=1.2
		]
		\node [atombody] (homme) at (0,-1.5) {homme$\backslash$1};
		\node [atomhead] (pere) at (3,0) {pere$\backslash$2};
		\node [atomhead] (hommeb) at (-0.2,0) {homme$\backslash$1};
		\node[termhead]	(z) at (1.5,0) {}
			edge[atomedge] node {1} (pere)
			edge[atomedge] node {1} (hommeb)
		;
		\node[termfr] (x) at (3,-1.5) {}
			edge[atomedge] node {1} (homme)
			edge[atomedge] node {2} (pere)
		;
		\node [] (zlabel) at (1.9,-0.4) {$z$};
		\node [] (xlabel) at (3.4,-1.9) {$x$};
		\atomlegend{-4}{0}
		\end{tikzpicture}
	\end{figure}
\end{frame}
